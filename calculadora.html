function calcularEconomia() { const hectares = parseFloat(document.getElementById('campoHectares').value); if (isNaN(hectares) || hectares <= 0) { alert("Por favor, insira um valor válido de hectares."); return; } // Base teórica: Economia estimada de 40.000 litros por hectare/dia const economiaLitrosDia = hectares * 40000; const economiaMes = economiaLitrosDia * 30; // Exibe o resultado na tela de forma amigável document.getElementById('resultado').innerHTML = `

Resultado da Economia Estimada:

Ao mudar para o gotejamento, sua propriedade economizará cerca de ${economiaLitrosDia.toLocaleString('pt-BR')} litros de água por dia!

No mês, isso equivale a uma preservação de ${economiaMes.toLocaleString('pt-BR')} litros.

`; }
